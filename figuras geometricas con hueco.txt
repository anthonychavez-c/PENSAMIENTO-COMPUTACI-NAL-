//figuras gemetrias con hueco
#include <iostream>
using namespace std;
int main() {
    int i,n,j,x;
    cin>>n;
    for (i=1; i<=n; i++)
    {
        cout<<endl;
        for (j=1; j<=n; j++){
         if (i == 1 or i == n or j == 1 or j == n){
                cout << " * "; 
            }
            else {
                cout << "   ";
        }
    }
    }
    return 0;
}
