#include <iostream>
using namespace std;
int main() {
     int a[3][3]={{0,1,2},{3,4,5},{6,7,8}};
     int b[3]={0};
     int c[3]={0};
    for(int i=0; i<3; i++){
       for(int j=0; j<3; j++){
        b[i]+=a[i][j];
       }
    }
     for(int i=0; i<3; i++){
       for(int j=0; j<3; j++){
           c[j]+=a[i][j];
       }
     }
      cout<<"la suma de fila "<<endl;
      for(int i=0; i<3; i++){
          cout<<b[i]<<" ";
      }
    cout<<endl;  
     cout<<"la suma de columna "<<endl;
     for(int j=0; j<3; j++){
         cout<<c[j]<<" ";
     }
    return 0;
}
