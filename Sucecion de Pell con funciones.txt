//Sucecion de Pell con funciones
#include <iostream>
using namespace std;
int Pell(int n){
    if(n==0){
        return 0;
    }
    else if(n==1){
        return 1;
    }
    else{
        return 2*Pell(n-1)+Pell(n-2);
    }
}
int main(){
    int n;
    cin>>n;
    cout<<"La sucecion de Pell es: "<<endl;
    for(int i=0;i<n;i++){
        cout<<Pell(i)<<", ";
    }
    return 0;
}
