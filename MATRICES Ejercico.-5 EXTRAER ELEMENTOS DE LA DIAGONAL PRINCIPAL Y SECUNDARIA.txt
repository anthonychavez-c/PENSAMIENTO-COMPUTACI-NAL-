#include <iostream>
using namespace std;
int main() {
     int a[3][3]={{0,1,2},{3,4,5},{6,7,8}};
     int n=3,DP,DS;
     cout<<"diagonal primaria: ";
    for(int i=0; i<3; i++){
      DP=a[i][i];
      cout<< " " <<DP;
    }
    cout<<endl;
    cout<<"diagonal secundaria: ";
      for(int i=0; i<3; i++){
         DS=a[i][n-i-1];
         cout<< " " <<DS;
      }
        cout<<endl;
    return 0;
}
