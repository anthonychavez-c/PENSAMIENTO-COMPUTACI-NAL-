// Codigo, nombre y calificacion con estructuras
#include <iostream>
#include <string>
using namespace std;
struct pensamiento{
  int codigo;
  string nombre;
  float calificacion;
};

int main(){
    int n;
    cin>>n;
    pensamiento persona[n];
    for (int i=0;i<n;i++){
        cout<<"Ingrese codigo de la persona "<<i+1<<" : ";
        cin>>persona[i].codigo;
        cout<<"Ingrese nombre de la persona "<<i+1<<" : ";
        cin>>persona[i].nombre;
        cout<<"Ingrese calificacion de la persona "<<i+1<<" : ";
        cin>>persona[i].calificacion;
        cout<<endl;
    }
    
    for(int i=0;i<n;i++){
        cout<<"\nEl codigo de la persona "<<i+1<<" es: "<<persona[i].codigo<<endl;
        cout<<"El nombre de la persona "<<i+1<<" es: "<<persona[i].nombre<<endl;
        cout<<"La calificacion de la persona "<<i+1<<" es: "<<persona[i].calificacion<<endl;
    }
    return 0;
}
