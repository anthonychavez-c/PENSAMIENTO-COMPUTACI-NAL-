// Suma de filas y columnas
#include <iostream>
using namespace std;

int main() {
    int A[3][3] = {{0,1,2},
                   {3,4,5},
                   {6,7,8}
        
    };
    int f=0;
    int c=0;
    for (int i = 0; i < 3; i++) {
            f=0;
        for (int j = 0; j < 3; j++) {
            f = f + A[i][j];
        }
        cout << "suma de filas     :" << f << endl;
    }
    for (int j = 0; j < 3; j++) {
            c=0;
        for (int i = 0; i < 3; i++) {
            c = c + A[i][j];
        }
        cout << "suma de columnas  :" << c << endl;
    }
    return 0;
}
