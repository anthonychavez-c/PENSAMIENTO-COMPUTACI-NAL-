// MATRICES Ejercico.-3 SUMA DE ELEMENTOS DE UNA MATRIZ
#include <iostream>
using namespace std;
int main() {
    int n;
     cout<<"ingrese n: ";
     cin>>n;
     int a[n][n];
    for(int i=0; i<n; i++){
       for(int j=0; j<n; j++){
        cin>>a[i][j];
       }
    }
    int suma = 0;
     for(int i=0; i<n; i++){
       for(int j=0; j<n; j++){
           suma+=a[i][j];
       }
     }
      cout<<"la suma es : "<<suma<<endl;
    return 0;
}
